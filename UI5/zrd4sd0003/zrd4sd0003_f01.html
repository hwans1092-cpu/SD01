FORM modify_screen .

  LOOP AT SCREEN.

    "── 대량 블록 ──────────────────────────────────
    IF screen-group1 = 'GR2'.
      screen-active = COND #( WHEN pa_b = 'X' THEN 1 ELSE 0 ).
      MODIFY SCREEN.
    ENDIF.

    "── 배치 전체 블록 ─────────────────────────────
    IF screen-group1 = 'GR3'.
      screen-active = COND #( WHEN pa_t = 'X'  THEN 1 ELSE 0 ).
      MODIFY SCREEN.
    ENDIF.

    "── 스케줄 라디오버튼 (배치 시 항상 표시) ──────
    IF screen-group1 = 'GR4'.
      screen-active = COND #( WHEN pa_t = 'X' THEN 1 ELSE 0 ).
      MODIFY SCREEN.
    ENDIF.

    "── 주기반복 전용 필드 ─────────────────────────
    IF screen-group1 = 'GR5'.
      screen-active = COND #(
        WHEN pa_t = 'X' AND pa_repet = 'X' THEN 1
        ELSE 0 ).
      MODIFY SCREEN.
    ENDIF.

    "── 1회예약 전용 필드 ──────────────────────────
    IF screen-group1 = 'GR6'.
      screen-active = COND #(
        WHEN pa_t = 'X' AND pa_once = 'X' THEN 1
        ELSE 0 ).
      MODIFY SCREEN.
    ENDIF.

    "── 완료알림/Commit (배치 시 항상 표시) ────────
    IF screen-group1 = 'GR7'.
      screen-active = COND #( WHEN pa_t = 'X'  OR pa_b = 'X' THEN 1 ELSE 0 ).
      MODIFY SCREEN.
    ENDIF.

  ENDLOOP.

ENDFORM.

FORM get_ddo7t_text USING pv_domname TYPE domname
                              pv_domvalue

  SELECT FROM dd07t
         FIELDS ddtext, domvalue_l
         WHERE domname    EQ @pv_domname
         AND   ddlanguage EQ @sy-langu
         INTO TABLE @DATA(lt_auart).

       WITH KEY domvalue_l = pv_domvalue.

  pv_ddtext = ls_auart-ddtext.

*& Form select_bill_target
*&---------------------------------------------------------------------*
FORM select_bill_target .

  SELECT FROM ztd4sd0029
         FIELDS auart,        " 판매문서유형
                description   " 텍스트
         WHERE spras EQ @sy-langu
         INTO TABLE @gt_auartt.

* 자재명 가져오기
  SELECT FROM ztd4mm0016
         FIELDS matnr,  " 자재번호
                maktx   " 자재명
         WHERE spras EQ @sy-langu
         INTO TABLE @gt_matnrt.

  REFRESH gt_bill_target.

  SELECT FROM ztd4sd0014 AS hdr
         INNER JOIN ztd4sd0015 AS itm
            ON itm~vbeln_de EQ hdr~vbeln_de
         INNER JOIN ztd4sd0008 AS soh
            ON soh~vbeln    EQ hdr~vbeln_so
         INNER JOIN ztd4sd0009 AS soi
            ON soi~vbeln    EQ hdr~vbeln_so    " 판매오더번호로 연결
           AND soi~matnr    EQ itm~matnr       " 자재번호로 매칭
         LEFT OUTER JOIN ztd4sd0020 AS ivm     " 청구아이템
            ON ivm~vbeln_de = hdr~vbeln_de
         LEFT OUTER JOIN ztd4sd0019 AS iv      " 청구헤더
            ON iv~vbeln_iv = ivm~vbeln_iv
         FIELDS hdr~vbeln_de,              " 출하문서번호
                hdr~kunnr,                 " 고객번호
                hdr~vbeln_so,              " 판매오더번호
                hdr~wadat_ist,             " PGI 일자
                hdr~lfdat,                 " 납품요청일
                itm~posnr      AS posnr_de," 출하아이템번호
                itm~matnr,                 " 자재번호
                itm~lfimg,                 " 출하수량
                itm~vrkme,                 " 단위
                soi~posnr      AS posnr_so," 판매오더 아이템번호
                soi~kbetr,                 " 단가
                soi~netwr,                 " 금액
                soi~disc_amt,              " 할인금액
                soi~waers,                 " 통화
                soh~auart,                 " 주문유형
                iv~fkdat,                  " 청구일자 - 있으면 기청구, 없으면 미청구
                iv~vbeln_iv                " 청구번호
         WHERE hdr~vbeln_de  IN @so_vbeln
           AND hdr~kunnr     IN @so_kunnr
           AND hdr~wadat_ist IN @so_wadat
           AND hdr~lfdat     IN @so_lfdat
           AND soi~vbeln     IN @so_vbso
           AND soh~auart     IN @so_auart
           AND itm~matnr     IN @so_matnr
           AND hdr~status    = 'PGI'      " PGI 완료된 건만
*          이미 청구된 건은 제외: 출하문서 단위로 체크
           AND NOT EXISTS (
               SELECT 1 FROM ztd4sd0020 AS iv
                WHERE iv~vbeln_de EQ hdr~vbeln_de
           )
         ORDER BY hdr~kunnr, hdr~vbeln_de, itm~posnr
         INTO CORRESPONDING FIELDS OF TABLE @gt_bill_target.

* Wa 초기화
  CLEAR gs_bill_target.

  " 청구일자 + 초기 아이콘 세팅
  LOOP AT gt_bill_target INTO gs_bill_target.

*   상태 아이콘 값 세팅
    PERFORM modify_icon_100.

*   텍스트 가져오기
    PERFORM modify_text_100.

    MODIFY gt_bill_target FROM gs_bill_target TRANSPORTING light
                                                           auart
                                                           matnrt.

  ENDLOOP.

  DATA(lv_count) = lines( gt_bill_target ).

  gv_title_100 = |청구 생성: { lv_count }건|.

ENDFORM.
